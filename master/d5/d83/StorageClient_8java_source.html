552  private ScanVertexResultIterator scanVertex(String spaceName,
553  List<Integer> parts,
554  String tagName,
555  List<String> returnCols,
556  boolean noColumns,
557  int limit,
558  long startTime,
559  long endTime,
560  boolean allowPartSuccess,
561  boolean allowReadFromFollower) {
562  if (spaceName == null || spaceName.trim().isEmpty()) {
563  throw new IllegalArgumentException("space name is empty.");
564  }
565  if (tagName == null || tagName.trim().isEmpty()) {
566  throw new IllegalArgumentException("tag name is empty");
567  }
568  if (noColumns && returnCols == null) {
569  throw new IllegalArgumentException("returnCols is null");
570  }
571 
572  Set<PartScanInfo> partScanInfoSet = new HashSet<>();
573  for (int part : parts) {
574  HostAddr leader = metaManager.getLeader(spaceName, part);
575  partScanInfoSet.add(new PartScanInfo(part, new HostAddress(leader.getHost(),
576  leader.getPort())));
577  }
578  List<HostAddress> addrs = new ArrayList<>();
579  for (HostAddr addr : metaManager.listHosts()) {
580  addrs.add(new HostAddress(addr.getHost(), addr.getPort()));
581  }
582 
583  long tag = metaManager.getTag(spaceName, tagName).getTag_id();
584  List<byte[]> props = new ArrayList<>();
585  props.add("_vid".getBytes());
586  if (!noColumns) {
587  if (returnCols.size() == 0) {
588  Schema schema = metaManager.getTag(spaceName, tagName).getSchema();
589  for (ColumnDef columnDef : schema.getColumns()) {
590  props.add(columnDef.getName());
591  }
592  } else {
593  for (String prop : returnCols) {
594  props.add(prop.getBytes());
595  }
596  }
597  }
598  VertexProp vertexCols = new VertexProp((int) tag, props);
599  List<VertexProp> vertexProps = Arrays.asList(vertexCols);
600  ScanVertexRequest request = new ScanVertexRequest();
601  request
602  .setSpace_id(getSpaceId(spaceName))
603  .setReturn_columns(vertexProps)
604  .setLimit(limit)
605  .setStart_time(startTime)
606  .setEnd_time(endTime)
607  .setEnable_read_from_follower(allowReadFromFollower);
608 
609  return doScanVertex(spaceName, tagName, partScanInfoSet, request, addrs, allowPartSuccess);
610  }

639  .withAddresses(addrs)
640  .withSpaceName(spaceName)
641  .withTagName(tagName)
642  .withPartSuccess(allowPartSuccess)
643  .build();
644  }
